    if (cp <= 0x7F) {
        out[0] = (unsigned char)cp;
        return 1;
    } else if (cp <= 0x7FF) {
        out[0] = (unsigned char)(0xC0 | (cp >> 6));
        out[1] = (unsigned char)(0x80 | (cp & 0x3F));
        return 2;
    } else if (cp <= 0xFFFF) {
